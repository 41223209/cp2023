#include <stdio.h>

int reverse(int n) {
    int d, y = 0;
    while (n) {
        d = n % 10;
        if ((n > 0 && y > (0x7fffffff - d) / 10) ||
            (n < 0 && y < ((signed)0x80000000 - d) / 10)) {
            return 0;  
        }
        y = y * 10 + d;
        n = n / 10;
    }
    return y;  
}

int main(void)
{
    int i = 123;
    printf("原始整数: %d  ",i);
    printf("\n反转后整数: %d  ",reverse(i));
    i = 208478933;
    printf("\n原始整数: %d  ",i);
    printf("\n反转后整数: %d  ",reverse(i));
    i = -73634;
    printf("\n原始整数: %d  ",i);
    printf("\n反转后整数: %d  ",reverse(i));
    return 0;
}

#include<stdio.h>
#include<stdlib.h>

int main ()
{
    char buffer[123]; // 定义一个字符数组 'buffer' 以存储用户输入
    unsigned long ul; // 定义一个无符号长整型变量 'ul' 以存储转换后的值

    // 打印一条消息，询问用户输入
    printf("\n请输入一个无符号数：");

    // 从标准输入读取用户输入，并将其存储在 'buffer' 中
    fgets(buffer, 123, stdin);

    // 使用基数0（自动检测）将 'buffer' 中的字符串转换为无符号长整型
    ul = strtoul(buffer, NULL, 0);

    // 打印转换后的值
    printf("输出：%lu\n\n", ul);

    return 0; // 表示程序成功执行
}

#include<stdio.h>      // 包含标准输入/输出头文件。
#include<stdlib.h>     // 包含标准库头文件。

int main ()          // 主函数开始。
{
    char input[] = "8.0 2.0";  // 声明一个字符数组 'input'，并用 "8.0 2.0" 初始化。
    char *ptr_end;           // 声明一个指向字符的指针 'ptr_end'。

    double x, y;              // 声明两个双精度浮点数变量 'x' 和 'y'。

    x = strtod(input, &ptr_end);   // 将 'input' 的第一部分转换为双精度浮点数，存储在 'x' 中，并更新 'ptr_end'。
    y = strtod(ptr_end, NULL);     // 将 'input' 的剩余部分转换为双精度浮点数，存储在 'y' 中。

    printf("\n输出= %.2lf\n\n", x / y);   // 打印 x/y 的结果，保留两位小数。

    return 0;   // 返回0以表示程序成功执行。
}   // 主函数结束。

#include<stdio.h>     // 包含标准输入/输出头文件。
#include<stdlib.h>    // 包含标准库头文件。
#include<time.h>      // 包含时间头文件以生成随机数。

int main ()           // 主函数的开始。
{
    int number, input;   // 声明两个整数变量 'number' 和 'input'。

    srand ( time(NULL) );   // 使用当前时间初始化随机数种子。

    number = rand() % 10 + 1;   // 生成一个介于1和10之间的随机数并将其存储在 'number' 中。

    do {   // 开始一个do-while循环。
        printf ("\n猜数字（1到10之间）：");   // 打印提示用户猜测数字的消息。
        scanf ("%d", &input);    // 读取用户的输入并将其存储在 'input' 中。

        if (number > input)   // 如果随机数大于用户的输入。
            printf ("数字更大\n");   // 打印消息指示数字更大。

    } while (number != input);   // 只要用户的输入不等于随机数，就继续循环。

    printf ("猜对了！\n\n");   // 打印消息指示用户猜对了。

    return 0;   // 返回0以表示程序成功执行。
}   // 主函数结束。
